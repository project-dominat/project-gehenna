Locate the section that draws the 8 lines (4 shadow + 4 color) and wrap them in a switch. Keep the existing Cross path identical; add three new branches.

```csharp
switch (shape)
{
    case CrosshairShape.Cross:
    default:
        // existing 8 DrawLine calls â€” unchanged
        DrawLine(new Vector2(-gap - length, 0f) + shadowOffset, new Vector2(-gap, 0f) + shadowOffset, shadow, outlineWidth);
        DrawLine(new Vector2(gap, 0f) + shadowOffset, new Vector2(gap + length, 0f) + shadowOffset, shadow, outlineWidth);
        DrawLine(new Vector2(0f, -gap - length) + shadowOffset, new Vector2(0f, -gap) + shadowOffset, shadow, outlineWidth);
        DrawLine(new Vector2(0f, gap) + shadowOffset, new Vector2(0f, gap + length) + shadowOffset, shadow, outlineWidth);
        DrawLine(new Vector2(-gap - length, 0f), new Vector2(-gap, 0f), color, lineWidth);
        DrawLine(new Vector2(gap, 0f), new Vector2(gap + length, 0f), color, lineWidth);
        DrawLine(new Vector2(0f, -gap - length), new Vector2(0f, -gap), color, lineWidth);
        DrawLine(new Vector2(0f, gap), new Vector2(0f, gap + length), color, lineWidth);
        break;

    case CrosshairShape.Dot:
        // Ring: draw filled shadow circle slightly larger, then filled color circle
        // to fake a stroked ring (DrawCircle has no stroke width â€” bool filled only)
        screen.DrawCircle(center + shadowOffset, gap + outlineWidth * 0.5f, shadow, false);
        screen.DrawCircle(center, gap, color, false);
        // Filled center dot
        screen.DrawCircle(center + shadowOffset, CenterDotOutlineRadius, shadow);
        screen.DrawCircle(center, CenterDotRadius, color);
        break;

    case CrosshairShape.Circle:
        // Ring only â€” shadow behind, color on top
        screen.DrawCircle(center + shadowOffset, gap + outlineWidth * 0.5f, shadow, false);
        screen.DrawCircle(center, gap, color, false);
        break;

    case CrosshairShape.Chevron:
        // Upper chevron: "V" shape (two diagonal lines meeting below)
        var chevArmLen = length * 1.2f;
        var chevAngle = MathF.PI / 5f; // 36 degrees from vertical
        var cx = MathF.Sin(chevAngle) * chevArmLen;
        var cy = MathF.Cos(chevAngle) * chevArmLen;
        // Top chevron: left arm and right arm
        DrawLine(new Vector2(-cx, -gap - cy) + shadowOffset, new Vector2(0f, -gap) + shadowOffset, shadow, outlineWidth);
        DrawLine(new Vector2(cx, -gap - cy) + shadowOffset, new Vector2(0f, -gap) + shadowOffset, shadow, outlineWidth);
        DrawLine(new Vector2(-cx, -gap - cy), new Vector2(0f, -gap), color, lineWidth);
        DrawLine(new Vector2(cx, -gap - cy), new Vector2(0f, -gap), color, lineWidth);
        // Bottom chevron: inverted
        DrawLine(new Vector2(-cx, gap + cy) + shadowOffset, new Vector2(0f, gap) + shadowOffset, shadow, outlineWidth);
        DrawLine(new Vector2(cx, gap + cy) + shadowOffset, new Vector2(0f, gap) + shadowOffset, shadow, outlineWidth);
        DrawLine(new Vector2(-cx, gap + cy), new Vector2(0f, gap), color, lineWidth);
        DrawLine(new Vector2(cx, gap + cy), new Vector2(0f, gap), color, lineWidth);
        break;
}
```

using Content.Shared.Weapons.Ranged.Events;
using Robust.Server.Player;
using Robust.Shared.Player;
using Robust.Shared.Projectiles;

namespace Content.Server.Weapons.Ranged.Systems;

public sealed class GunHitFeedbackSystem : EntitySystem
{
    [Dependency] private readonly IPlayerManager _playerManager = default!;

    public override void Initialize()
    {
        base.Initialize();

        SubscribeLocalEvent<HitscanBasicDamageComponent, HitscanDamageDealtEvent>(OnHitscanDamage);
        SubscribeLocalEvent<ProjectileComponent, ProjectileDamageDealtEvent>(OnProjectileHit);
    }

    private void OnHitscanDamage(Entity<HitscanBasicDamageComponent> ent, ref HitscanDamageDealtEvent args)
    {
        if (args.DamageDealt.GetTotal() <= 0)
            return;
        if (args.Shooter == null)
            return;

        NotifyShooter(args.Shooter.Value);
    }

    private void OnProjectileHit(Entity<ProjectileComponent> ent, ref ProjectileDamageDealtEvent args)
    {
        // DamageDealt is post-resistance (from TryChangeDamage out param) â€” guaranteed > 0 by caller
        if (args.Shooter == null)
            return;

        NotifyShooter(args.Shooter.Value);
    }

    private void NotifyShooter(EntityUid shooter)
    {
        if (!_playerManager.TryGetSessionByEntity(shooter, out var session))
            return; // NPC or no session â€” silent skip (confirmed pattern in codebase)

        // session.Channel â€” confirmed pattern (e.g. ExamineSystem, GameTicker)
        RaiseNetworkEvent(new GunHitConfirmedEvent(), session.Channel);
    }
}
